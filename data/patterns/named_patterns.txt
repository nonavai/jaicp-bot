# Именованные паттерны для автосервисного чат-бота

# Паттерн для полного имени (ФИО)
FullName = ([А-Яа-яA-Za-z]+ [А-Яа-яA-Za-z]+( [А-Яа-яA-Za-z]+)?)

# Паттерн для номера телефона (российский формат)
Phone = (\+?[78]?[\d\-\s\(\)]{10,}|[\d\-\s\(\)]{10,})

# Паттерн для марки автомобиля
CarBrand = (Skoda|Lada|Toyota|KIA|Hyundai|Volkswagen|BMW|Mercedes|Audi|Ford|Chevrolet|Nissan|Mazda|Honda|Renault|Peugeot|Citroen|Opel|Volvo|Lexus|Infiniti|Acura|Subaru|Mitsubishi|Suzuki|Daihatsu|Fiat|Alfa Romeo|Seat|Skoda|Škoda)

# Паттерн для записи на ТО
ServiceRequest = (хочу записаться на ТО|запиши меня на техобслуживание|подошло время ТО|пройти ТО|записаться на сервис|нужно ТО|техобслуживание|техосмотр|диагностика|ремонт)

# Паттерн для приветствия
Greeting = (привет|здравствуйте|добрый день|доброе утро|добрый вечер|здравствуй)

# Паттерн для прощания
Farewell = (до свидания|пока|увидимся|до встречи|спасибо|спасибо за помощь)

# Паттерн для подтверждения
Confirmation = (да|конечно|разумеется|точно|именно|правильно|верно|ага|угу)

# Паттерн для отрицания
Negation = (нет|не|неправильно|неверно|ошибка|не то|не так)

# Паттерн для уточнения
Clarification = (что|как|где|когда|почему|зачем|какой|какая|какое|какие)

# Паттерн для времени
Time = (сегодня|завтра|вчера|утром|днем|вечером|ночью|сейчас|потом|позже|раньше)

# Паттерн для цены
Price = (сколько стоит|цена|стоимость|тариф|расценки|оплата|плата)
